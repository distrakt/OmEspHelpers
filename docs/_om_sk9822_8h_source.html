35 // send N bits down the wire, remembering that it's MSB first.
36 void wN(unsigned long value, int bitCount)
37 {
38 unsigned long mask = 1L << (bitCount - 1);
39 while (bitCount--)
40 {
41 int bit = (value & mask) ? 1 : 0;
42 w1(bit);
43 value <<= 1;
44 }
45 }

65 void w32(unsigned long x)
66 {
67 if(USE_SPI)
68 {
69 SPI.transfer (x >> 24);
70 SPI.transfer (x >> 16);
71 SPI.transfer (x >> 8);
72 SPI.transfer (x >> 0);
73 ww += x;
74 }
75 else
76 {

78 {
79 DIGITAL_WRITE(CLOCK_PIN, 0);

102 void showStrip(OmLed16Strip *strip)

121 w32(strip->leds[ix].dot());
122 w32(0);
