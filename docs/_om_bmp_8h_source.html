38 bool putBmpHeader(int width, int height)
39 {
40 int rowBytes = ((width + 7) / 8 + 3) & 0xfffffffc;
41 int dataSize = rowBytes * height;
42 int paletteSize = 8; // 2 colors, 4 bytes each.
43 const int bmpHeaderSize = 14;
44 const int bmpInfoHeaderSize = 40;
45 int fileSize = bmpHeaderSize + bmpInfoHeaderSize + paletteSize + dataSize;
46
47 bool result = true;
48 // header
49 result &= put8('B');
50 result &= put8('M');
51 result &= put32(fileSize);
52 result &= put16(0); // reserved1
53 result &= put16(0); // reserved2
54 result &= put32(bmpHeaderSize + bmpInfoHeaderSize + paletteSize); // offset to data
55
56 // DIB info header
57 result &= put32(bmpInfoHeaderSize);
58 result &= put32(this->width); // width
59 result &= put32(this->height); // height
60 result &= put16(1); // planes
61 result &= put16(1); // bits
62 result &= put32(0); // compression
63 result &= put32(dataSize);
64 result &= put32(1000); // xresolution px/m
65 result &= put32(1000); // yresolution
66 result &= put32(2); // ncolours
67 result &= put32(0); // important colors
68
69 // Write the two color palette.
70
71 result &= put32(0x00101010); // color black. (not really black and white, for my thing this looks good)
72 result &= put32(0x00c0c0c0); // color white.
73 return result;
74 }

95 bool put1Bit(int bit)
96 {
97 if(this->x >= this->width && this->y >= this->height)
98 return false;
99 bool result = true;
100
101 if(!didPutHeader)
102 {
103 result &= this->putBmpHeader(this->width, this->height);
104 this->didPutHeader = true;
105 }
106 this->byteInProgress |= bit << (7 - this->x % 8);
107 this->x++;
108 if(this->x == this->width || (this->x & 0x0007) == 0) // finished one byte in progress
109 {
110 this->consumer->put(this->byteInProgress);
111 this->byteInProgress = 0;
112 }
113
114 if(this->x == this->width)
115 {
116 // at the end of the row, do our special dance
117 for(int ix = 0; ix < this->rowPadBytes; ix++)
118 result &= this->put8(0);
119 this->x = 0;
120 this->y++;
121 }
122 return result;
123 }
124
125
126private:
127 bool putN(int n, uint8_t *data)
128 {
129 bool result = true;
130 while(n-- > 0)
131 result &= this->consumer->put(*data++);
132 return result;
133 }

219 bool put1Pixel(uint8_t r, uint8_t g, uint8_t b)
220 {
221 if(this->x >= this->width && this->y >= this->height)
222 return false;
223 bool result = true;
224
225 if(!didPutHeader)
226 {
227 result &= this->putBmpHeader();
228 this->didPutHeader = true;
229 }
230 result &= this->put8(b);
231 result &= this->put8(g);
232 result &= this->put8(r);
233
234 this->x++;
235 if(this->x >= this->width)
236 {
237 // Each row must be padded to a multiple of four BYTES.
238 // since each pixel is 3 bytes, this can happen.
239 int extras = (4 - (this->width * 3) % 4) % 4;
240 while(extras-- > 0)
241 this->put8(0);
242
243 this->x = 0;
244 this->y++;
245 // And did we just hit the final row?
246 if(this->y >= this->height)
247 {
248 this->consumer->done();
249 this->done();
250 }
251 }
252
253 return result;
254 }

291public:
292 OmIByteStream *consumer = NULL;

294 OmBase64Stream(OmIByteStream *consumer)
295 {
296 this->consumer = consumer;
297 }
298
299 uint8_t buf[3];
300 int bufSize = 0;
301 bool put(uint8_t ch) override
302 {
303 if(this->isDone)
304 return false;
305
306 bool result = true;
307 if(this->bufSize < 3)
308 this->buf[this->bufSize++] = ch;
309
310 if(this->bufSize == 3)
311 {
312 // when we get three chars, emit four.
313 result &= this->consumer->put(base64_table[this->buf[0] >> 2]);
314 result &= this->consumer->put(base64_table[((this->buf[0] & 0x03) << 4) | (this->buf[1] >> 4)]);
315 result &= this->consumer->put(base64_table[((this->buf[1] & 0x0f) << 2) | (this->buf[2] >> 6)]);
316 result &= this->consumer->put(base64_table[this->buf[2] & 0x3f]);
317 bufSize = 0;
318 }
319 return result;
320 }
321
322 bool done() override
323 {
324 bool result = true;
325 this->isDone = true;
326 // close any last bits
327 switch(this->bufSize)
328 {
329 case 0:
330 break;
331
332 case 1:
333 result &= this->consumer->put(base64_table[this->buf[0] >> 2]);
334 result &= this->consumer->put(base64_table[(this->buf[0] & 0x03) << 4]);
335 result &= this->consumer->put('=');
336 result &= this->consumer->put('=');
337 break;
338
339 case 2:
340 result &= this->consumer->put(base64_table[this->buf[0] >> 2]);
341 result &= this->consumer->put(base64_table[((this->buf[0] & 0x03) << 4) | (this->buf[1] >> 4)]);
342 result &= this->consumer->put(base64_table[(this->buf[1] & 0x0f) << 2]);
343 result &= this->consumer->put('=');
344 break;
345 }
346 this->bufSize = 0;
347 return result;
348 }
