155 void fill2(LEDT co, float x0, float y0, float x1, float y1)
156 {

172 int y1i = floor(y1);
173

177 float f = y1 - y0;
178 LEDT coF = co * f;
179 this->fillRow<0>(coF, y0i, x0, x1);
180 }
181 else
182 {
183 LEDT coF;
184 float f;
185
186 // topmost row
187 f = y0i + 1 - y0;
188 coF = co * f;
189 this->fillRow<0>(coF, y0i, x0, x1);
190
191 // middle rows, if any
192 for (int k = y0i + 1; k < y1i; k++)
193 {
194 this->fillRow<0>(co, k, x0, x1);
195 }
196
197 // bottom pixel row
198 f = y1 - y1i;
199 coF = co * f;
200 this->fillRow<0>(coF, y1i, x0, x1);
201 }
