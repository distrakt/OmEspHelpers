198 static int migrate(int currentValue, int targetValue, float portion)
199 {
200 float distance = fabs((float)targetValue - (float)currentValue);
201 float migrateAmountF = distance * portion;
202 int migrateAmountI = migrateAmountF;
203 float migrateAmountFract = migrateAmountF - migrateAmountI;
204 // stochastically handle the fraction.
205 if(rr(1.0) < migrateAmountFract)
206 migrateAmountI ++;
207 // redo the newvalue
208 int newValue = migrateI(currentValue, targetValue, migrateAmountI);
209
210 return newValue;
211 }

221 void tick(unsigned int ms, STRIPT *ledStrip)
222 {
223 // ms is how far addvanced.
224 this->tickStats(ms);
225
226 if(this->crossfadeMs)
227 {
228// printf("fading %d of %d\n", this->crossfadeMsSoFar, this->crossfadeMs);
229 // we are crossfading! so be it.
230 if(this->crossfadeParamCount)
231 {
232 // newfangled param morphing
233 this->crossfadeMsSoFar += ms;
234 PATTERNT *cp = this->currentPattern;
235 if(this->crossfadeMsSoFar >= this->crossfadeMs)
236 {
237 // we're done crossfading. we have arrived.
238 cp->setParams(this->crossfadeFirstParam,
239 this->crossfadeParamCount,
240 this->crossfadeParamsTarget);
241// printf("Y done interpolating #%06x, #%06x\n", this->crossfadeParamsTarget[0], this->crossfadeParamsTarget[1]);
242 this->crossfadeParamCount = 0;
243 this->crossfadeMs = 0;
244 this->crossfadeMsSoFar = 0;
245 goto singlePattern;
246 }
247
248 // ok yes still crossfade ing
249 for(int ix = this->crossfadeFirstParam; ix < this->crossfadeFirstParam + this->crossfadeParamCount; ix++)
250 {
251 if(ix < 0 || ix >= cp->params.size())
252 break;
253
254 EOmLedPatternParamType paramType = cp->getParamType(ix);
255 uint32_t v0 = this->crossfadeParamsStart[ix - this->crossfadeFirstParam];
256 uint32_t v = this->crossfadeParamsTarget[ix - this->crossfadeFirstParam];
257
258 // if morphing params,
259 // the amount to migrate, from where we are, is
260 // tick-milliseconds * distance Ã· remaining-milliseconds
261 // precompute the time portion. distance is per.
262
263 float fractionThere = (float)this->crossfadeMsSoFar / this->crossfadeMs;
264
265 switch(paramType)
266 {
267 case PPT_INT:
268 {
269 int newValue = mapRange(fractionThere, 0.0, 1.0, v0, v) + 0.5;
270 cp->setParamValueInt(ix, newValue);
271 break;
272 }
273 case PPT_COLOR:
274 {
275 LEDT vCo = LEDT(v);
276 LEDT v0Co = LEDT(v0);
277 LEDT newValue;
278
279 newValue.r = mapRange(fractionThere, 0.0, 1.0, v0Co.r, vCo.r);
280 newValue.g = mapRange(fractionThere, 0.0, 1.0, v0Co.g, vCo.g);
281 newValue.b = mapRange(fractionThere, 0.0, 1.0, v0Co.b, vCo.b);
282 cp->setParamValueColor(ix, newValue);
283
284// printf("X interp #%06x - #%06x (%5.2f): #%06x\n", v0, v, fractionThere, newValue.toHex());
285 break;
286 }
287 default:
288 break;
289 }
290 }
291 // and before we go... tick the actual pattern, for pixels!
292 this->tickInto(ms, ledStrip, this->currentPattern);
293 }
294 else
295 {
296 // visual brightness fade
297 this->crossfadeMsSoFar += ms;
298 if(this->crossfadeMsSoFar >= this->crossfadeMs)
299 {
300 // we're done crossfading. we have arrived.
301 this->crossfadeMs = 0;
302 this->crossfadeMsSoFar = 0;
303 goto singlePattern;
304 }
305
306 // ok yes crossfade
307 if(!this->crossfadeStrip)
308 {
309 // could reallocate the strip if the ledCount differs, or ringLed0
310 this->crossfadeStrip = new STRIPT(this->ledCount);
311 this->crossfadeStrip->ringLed0 = ledStrip->ringLed0;
312 }
313
314 float t = (float)this->crossfadeMsSoFar / this->crossfadeMs;
315 this->tickInto(ms, ledStrip, this->currentPattern);
316 this->tickInto(ms, crossfadeStrip, crossfadePattern);
317
318 *ledStrip *= t;
319 *crossfadeStrip *= 1.0 - t;
320 *ledStrip += crossfadeStrip;
321 }
322 }
323 else
324 {
325 singlePattern:
326 this->tickInto(ms, ledStrip, this->currentPattern);
327 }
328 }

330 void getPerformanceInfo(int &millisecondsPerFrameOut, int &framesPerSecondOut)
331 {
332 float result = 0;
333 for (int ix = 0; ix < PM_HISTORY_SIZE; ix++)
334 {
335 result += msHistory[ix];
336 }
337 result = result / PM_HISTORY_SIZE;
338 if (result < 1)
339 result = 1;
340 millisecondsPerFrameOut = result + 0.5;
341 framesPerSecondOut = (1000.0 / result) + 0.5;
342 }
